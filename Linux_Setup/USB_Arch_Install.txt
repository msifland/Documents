--Create usb arch install from existing Arch system to usb drive.

-Try this method first:
-Copy aif-master to root directory, open terminal and type:
	su
	/aif-master/install.sh
	-Prepare and mount usb drive following installer instructions.
	-Set host name to arch-usb.

-When installer reaches 'mkinitcpio' open new terminal and type:
	sudo geany /mnt/etc/mkinitcpio.conf
	-Move the block hook in /mnt/etc/mkinitcpio.conf directly behind the udev hook: HOOKS="base udev block ...", so the initial ramdisk environment can boot your kernel from the flash drive.

	-Then continue and go all the way through the install process in accordance with the MSI_Arch_Install.txt.
	-Install Unknown/Generic display drivers.
	-For desktop environment, install lxde, very fast light full desktop.
	-Install all Wireless Device Packages.
	-Install NetworkManager.
	-Install cups AND samba for printing.
	-Install .config directory only if larger usb drive(16g or more).
	-Close installer
	-Exit
		exit

-When finished, exit installer to terminal and reinstall grub for usb settings:
	sudo mount /dev/sdX# /
	-If you want a recovery option in grub then run:
		sudo geany /mnt/etc/default/grub
	-Comment out this line:
	GRUB_DISABLE_RECOVERY=true
	sudo arch-chroot /mnt
	grub-install --target=i386-pc --verbose --force --removable --recheck /dev/sdX - (may have to run twice if errors)
	grub-mkconfig -o /boot/grub/grub.cfg - (ignore lvmetad waring)
	exit
	-Unmount
		sudo umount /dev/sdX#

-Reboot
-Download backup from Dropbox or grab from existing system if on that pc. Copy to usb home directory. Menu/reboot/do not save sessions.

######## Manual Installation #########
-You can use gparted to erase, create partition table, and format usb.
-Prepare installation, open terminal and type:
	sudo mkfs.ext4 -L usbboot /dev/sdb1
	sudo mount /dev/sdb1 /mnt
	sudo pacstrap /mnt base base-devel
	sudo genfstab -U -p /mnt >> /mnt/etc/fstab

-Move the block hook in /mnt/etc/mkinitcpio.conf directly behind the udev hook: HOOKS="base udev block ...", so the initial ramdisk environment can boot your kernel from the flash drive:
	sudo geany /mnt/etc/mkinitcpio.conf

-Change root to /mnt/archusbdrive:
	sudo arch-chroot /mnt

-Set the desired hostname:
	echo archusb > /etc/hostname

-Set your timezone:
	ln -s /usr/share/zoneinfo/America/Chicago /etc/localtime

-Uncomment your locales in /etc/locale.gen and run locale-gen afterwards:
	nano /etc/locale.gen
	locale-gen

-Sset your preferred locale, g.e.:
	echo LANG=en_US.UTF-8 > /etc/locale.conf

-Create the initial ramdisk:
	mkinitcpio -p linux

-Set your root password:
	passwd

-Install a bootloader, for GRUB:
	pacman -S grub wireless_tools wpa_supplicant wpa_actiond dialog networkmanager
	grub-install --target=i386-pc --verbose --force --removable --recheck /dev/sdb
	grub-mkconfig -o /boot/grub/grub.cfg

-Exit the chroot, unmount the partitions umount /dev/sdX... and reboot your system:
	exit
	sudo umount /dev/sdb1
	sudo reboot

-Enable internet
	-If on wifi then:
		wifi-menu
		systemctl enable NetworkManager
	-If wired connection
		systemctl start dhcpcd.service
	-Update pacman
-Edit pacman.conf to include multilib and add archlinufr
	pacman -Syyu

-Install graphical packages:
	pacman -S xorg-drivers xorg-server xorg-xinit xorg-server-utils mesa
	pacman -S xfce4 xfce4-goodies
	pacman -S lightdm lightdm-gtk-greeter
	pacman -S nemo thunar kate file-roller xarchiver nemo-fileroller git ntp chromium geany
	systemctl enable lightdm ntpd

-Add user:
	useradd -m -g users -G wheel,storage,power,sys,lp -s /bin/bash msifland
	passwd msifland
	pacman -S sudo
	pacman -S xorg-twm xorg-xclock xterm

-Uncomment sudoers:
	geany /etc/sudoers
	-Uncomment # %wheel ALL(ALL) ALL
	-Add at bottom:
		Defaults timestamp_timeout=-1

-Reboot
